返回undefined的典型例子

    // 变量声明了，但没有赋值
    var i; // undefined


    // 调用函数时，应该提供的参数没有提供，该参数等于 undefined
    function f(x) {
    return x;
    }
    f(); // undefined


    // 函数没有返回值时，默认返回 undefined
    function f() {}
    f() // undefined
